        // 财位计算逻辑

        function calculateWealthCorner() {

            const door = document.getElementById('door').value;

            const resultDiv = document.getElementById('result');

            const locationSpan = document.getElementById('location');

            const tipsList = document.getElementById('tips');


            let wealthDirection = "";

            let tips = [];


            const solutions = {

                east: {

                    direction: "Southeast Corner",

                    tips: ["Place money plant", "Use purple decor", "Keep area clutter-free"]

                },

                south: {

                    direction: "Northwest Corner",

                    tips: ["Add metal elements", "Display coins", "Use white colors"]

                },

                west: {

                    direction: "Northeast Corner",

                    tips: ["Place crystals", "Use yellow tones", "Add lighting"]

                },

                north: {

                    direction: "Southwest Corner",

                    tips: ["Use earth tones", "Display pottery", "Add square shapes"]

                }

            };


            if (solutions[door]) {

                wealthDirection = solutions[door].direction;

                tips = solutions[door].tips;

            }


            locationSpan.textContent = wealthDirection;

            tipsList.innerHTML = tips.map(tip => `<li>${tip}</li>`).join('');

            resultDiv.style.display = 'block';

            resultDiv.scrollIntoView({behavior: 'smooth'});

        }

        }
